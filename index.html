function atualizarTopBar() { const nomeSalvo = localStorage.getItem('usuario_nome') || 'Visitante'; const tipoSalvo = localStorage.getItem('usuario_tipo'); if (nomeSalvo) { const topBar = document.getElementById('topBar'); const btnCadastrar = document.getElementById('btnCadastrar'); const btnEntrar = document.getElementById('btnEntrar'); if (btnCadastrar) btnCadastrar.style.display = 'none'; if (btnEntrar) btnEntrar.style.display = 'none'; if (!document.getElementById('userInfoDiv')) { const userDiv = document.createElement('div'); userDiv.id = 'userInfoDiv'; userDiv.className = 'user-info'; let painelLink = ''; if (tipoSalvo === 'ambos' || tipoSalvo === 'vendedor') { painelLink = ` Meu Painel `; } userDiv.innerHTML = ` ${painelLink} Olá, ${nomeSalvo} Sair `; topBar.appendChild(userDiv); } } }
